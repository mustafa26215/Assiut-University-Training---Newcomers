#include <iostream>
using namespace std;

int main() {
    long long a;
    cin >> a;

    for (int num = 2; num <= a; num++) {  
        long long Counter = 0;

        for (int i = 1; i <= num; i++) {  
            if (num % i == 0) {
                Counter++;
            }
        }

        if (Counter == 2) {  
            cout << num << " ";
        }
    }

    cout << endl;
    return 0;
}
