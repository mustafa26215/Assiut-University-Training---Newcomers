#include <iostream>
using namespace std;

int main() {
    int A, B;
    cin >> A >> B;

    bool found = false; 
    for (int i = A; i <= B; i++) {
        int temp = i;
        bool lucky = true; 

        while (temp > 0) {
            int digit = temp % 10; 
            if (digit != 4 && digit != 7) {
                lucky = false;
                break;
            }
            temp /= 10; 
        }

        if (lucky) {
            cout << i << " ";
            found = true;
        }
    }

    if (!found) cout << -1; 

    cout << endl;
    return 0;
}
