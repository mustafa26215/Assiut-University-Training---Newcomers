#include <iostream>
using namespace std;

int main() {
    int T;
    cin >> T;

    while (T--) {
        int N;
        cin >> N;

        int cnt = 0;

        while (N > 0) {
            if (N & 1) cnt++; // لو آخر bit = 1
            N >>= 1; // shift right
        }

        int result = (1 << cnt) - 1;
        cout << result << endl;
    }

    return 0;
}
