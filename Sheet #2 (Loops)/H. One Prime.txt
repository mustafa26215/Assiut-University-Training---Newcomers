#include <iostream>
using namespace std;
int main()
{
    long long a;
    cin>>a;
    long long Counter=0;
    for(int i=1;i<=a;i++){
        if(a%i==0){
            Counter=Counter+1;
        }
    }
    if(Counter==2){
        cout<<"YES"<<endl;
    }
    else{
        cout<<"NO"<<endl;
    }
    return 0;
}
