#include <iostream>
using namespace std;

int main()
{
    int Num;
    cin >> Num;

    bool found = false;

    for(int i = 1; i <= Num; i++){
        if(i % 2 == 0){
            cout << i << endl;
            found = true;
        }
    }

    if(!found){
        cout << -1 << endl;
    }

    return 0;
}
