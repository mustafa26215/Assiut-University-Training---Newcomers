#include <iostream>
using namespace std;
 
int main() {
    long long A, B, C;
    char S, equalSign;
 
    cin >> A >> S >> B >> equalSign >> C;
 
    long long result;
 
    if (S == '+') result = A + B;
    else if (S == '-') result = A - B;
    else result = A * B;
 
    if (result == C)
        cout << "Yes";
    else
        cout << result;
 
    return 0;
}
