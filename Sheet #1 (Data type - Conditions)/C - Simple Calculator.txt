/*
C. Simple Calculator
"X + Y = summation result" without quotes.
"X * Y = multiplication result" without quotes.
"X - Y = subtraction result" without quotes.
*/
#include <iostream>
using namespace std;
int main(){
    long long a,b;
    cin>>a>>b;
    cout<<a<<" + "<<b<<" = "<<a+b<<endl;
    cout<<a<<" * "<<b<<" = "<<a*b<<endl;
    cout<<a<<" - "<<b<<" = "<<a-b<<endl;
 
}
